The Social Media Downloader API works in any browser environment with no special configuration. CORS is open for all origins, so you can call the API directly from client-side JavaScript using `fetch` or `axios`. For most platforms, you get back a direct CDN URL and can trigger the download without touching your own server. TikTok is the exception — its CDN blocks direct browser requests, so those downloads are routed through the `/api/stream` proxy automatically.

**Base URL:** `https://fastapi-u8bm.onrender.com`

<Steps>
  <Step title="Fetch metadata to display title and thumbnail">
    Before downloading, call `POST /api/metadata` to retrieve video info for display in your UI.

    ```javascript
    const API_URL = 'https://fastapi-u8bm.onrender.com';

    async function fetchMetadata(videoUrl) {
      const response = await fetch(`${API_URL}/api/metadata`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ url: videoUrl }),
      });

      if (!response.ok) {
        const err = await response.json();
        throw new Error(err.detail || 'Failed to fetch metadata');
      }

      const { data } = await response.json();
      // data: { title, thumbnail, duration, uploader, platform, url }
      return data;
    }
    ```

    The response includes `title`, `thumbnail`, `duration`, `uploader`, and `platform`. Use these to render a preview card before the user confirms the download.
  </Step>

  <Step title="Get the proxied stream URL">
    Call `POST /api/get-direct-url` to receive the URL your browser should download from. The API returns either a direct CDN URL or a `/api/stream` proxy URL depending on the platform.

    ```javascript
    async function getDirectUrl(videoUrl) {
      const response = await fetch(`${API_URL}/api/get-direct-url`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ url: videoUrl }),
      });

      if (!response.ok) {
        const err = await response.json();
        throw new Error(err.detail || 'Failed to get download URL');
      }

      const { data } = await response.json();
      // data: { direct_url, filename, filesize, force_backend_stream, ... }
      return data;
    }
    ```

    When `force_backend_stream` is `true`, the `direct_url` already points to `/api/stream` — you do not need to route differently based on platform.
  </Step>

  <Step title="Download the video">
    Use `force_backend_stream` to decide how to trigger the download:

    - **`force_backend_stream: true`** (TikTok and similar) — the response is a streamed binary, so use XHR to track download progress as bytes arrive.
    - **`force_backend_stream: false`** (YouTube, Instagram, others) — create an `<a>` tag pointing to the `direct_url` and click it programmatically.

    ```javascript
    async function downloadVideo(videoUrl) {
      const { direct_url, filename, force_backend_stream } = await getDirectUrl(videoUrl);

      if (force_backend_stream) {
        // Use XHR for progress tracking on proxied streams
        return new Promise((resolve, reject) => {
          const xhr = new XMLHttpRequest();
          xhr.open('GET', direct_url, true);
          xhr.responseType = 'blob';

          xhr.onprogress = (event) => {
            if (event.lengthComputable) {
              const percentage = (event.loaded / event.total) * 100;
              console.log(`Downloaded: ${percentage.toFixed(1)}%`);
              // Update your progress bar here
            } else {
              // TikTok CDN may use chunked encoding — show received MB instead
              const mb = (event.loaded / 1024 / 1024).toFixed(1);
              console.log(`Downloaded: ${mb} MB`);
            }
          };

          xhr.onload = () => {
            if (xhr.status !== 200) {
              return reject(new Error(`Download failed: ${xhr.status}`));
            }
            const blob = new Blob([xhr.response], { type: 'video/mp4' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = filename || 'video.mp4';
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            URL.revokeObjectURL(url);
            resolve();
          };

          xhr.onerror = () => reject(new Error('Network error during download'));
          xhr.send();
        });
      } else {
        // Direct CDN URL — trigger download via <a> tag
        const a = document.createElement('a');
        a.href = direct_url;
        a.download = filename || 'video.mp4';
        document.body.appendChild(a);
        a.click();
        document.body.removeChild(a);
      }
    }
    ```
  </Step>
</Steps>

## CORS

The API sets `Access-Control-Allow-Origin: *`, so you can call it from any origin — including `localhost` during development. No proxy, no configuration needed.

## TikTok downloads

TikTok's CDN enforces strict `Referer` and `User-Agent` header checks that block direct browser requests. When you call `/api/get-direct-url` for a TikTok URL, the API always returns a `direct_url` pointing to `/api/stream` with `force_backend_stream: true`. The server acts as a transparent byte-proxy, forwarding the CDN response to your browser in 64 KB chunks.

Because `/api/stream` forwards the CDN's `Content-Length` header when available, `xhr.onprogress` will have `event.lengthComputable = true` for most TikTok downloads. When the CDN uses chunked transfer encoding (no `Content-Length`), `event.lengthComputable` is `false` — display received megabytes instead of a percentage in that case.
